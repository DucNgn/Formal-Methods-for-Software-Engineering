\subsection{Description}

\[ R = \{ (0, 0),  (0, 1),  (0, 3),  (1, 0),  (1, 1),  (2, 2),  (3, 0),  (3, 1), (3, 3), (4, 0), (4, 1), (4, 3), (4, 4)  \}\]

\[ U = \{ (0, 0),  (0, 1), (0, 3), (1, 0), (1, 1), (1, 3), (2, 2), (3, 0), (3, 1), (3, 3), (4, 4) \}\]
